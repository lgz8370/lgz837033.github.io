# 2. 两个时间相差多少天
/**
     * 获取两个时间相差多少天
     *
     * @param dates
     * @return
     */
    public static long getDiffDays(String[] dates) {
        if (dates.length < 2) {
            return 1;
        }
        DateTimeFormatter fmt = DateTimeFormatter.ofPattern("yyyy-MM-dd");
        LocalDate begin = LocalDate.parse(dates[0], fmt);
        LocalDate end = LocalDate.parse(dates[1], fmt);
        return end.toEpochDay() - begin.toEpochDay() + 1;
    }

# 3. 获取连续时间集合
 /**
     * 数据填充
     * 可操作: 天/周/月/年
     * @param time  间隔天数
     * @param begin 开始时间
     * @return
     */
    public static List<String> getRangeIncres2(long time, String begin) {

        List<String> list = Lists.newArrayList();
        DateTimeFormatter fmt = DateTimeFormatter.ofPattern("yyyy-MM-dd");
        LocalDate localDate = LocalDate.parse(begin, fmt);
        for (int i = 0; i < time; i++) {
            list.add(localDate.plusDays(i).toString());
        }
        return list;
    }

# 4. 计算时间差:xx天xx小时xx分钟
 /**
     * 计算时间差:xx天xx小时xx分钟
     * @param endDate 结束时间
     * @param beginDate 开始时间
     * @return
     */
    public static List<Long> getDatePoor(Date endDate, Date beginDate) {
        long nd = 1000 * 24 * 60 * 60;//每天毫秒数
        long nh = 1000 * 60 * 60;//每小时毫秒数
        long nm = 1000 * 60;//每分钟毫秒数
        long diff = endDate.getTime() - beginDate.getTime(); // 获得两个时间的毫秒时间差异
        long day = diff / nd;   // 计算差多少天
        long hour = diff % nd / nh; // 计算差多少小时
        long min = diff % nd % nh / nm;  // 计算差多少分钟
        System.out.println(day + "天" + hour + "小时" + min + "分钟");
        List<Long> list = new ArrayList<>();
        list.add(day);
        list.add(hour);
        list.add(min);
        return list;
    }
