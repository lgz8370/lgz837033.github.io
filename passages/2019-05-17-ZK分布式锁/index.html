  • 代码实现
/**
 * @author LuGuoZheng
 * @date 2019/5/15 13:33
 * @Version 3.0
 * @description
 */
public class ZKLock {

    /**
     * zk客户端
     */
    public static CuratorFramework client = null;

    public InterProcessMutex mutex = null;

    /**
     * 初始化zk客户端
     */
    static {
        System.out.println("开始初始化zk客户端");
        // 重试策略:重试3次,每次重试间隔1s
        RetryPolicy retryPolicy = new ExponentialBackoffRetry(1000, 3);
        client =
                CuratorFrameworkFactory.builder()
                        .connectString("139.159.253.188:2181")// zk地址
                        .sessionTimeoutMs(15000) // 会话超时时间
                        .connectionTimeoutMs(30000) // 连接超时时间
                        .retryPolicy(retryPolicy) // 重试策略
                        .namespace("YL_LGZ") // 名称
                        .build();

        // 启动zk客户端连接zk
        client.start();
    }

    /**
     *
     * @param code 节点名称
     */
    public void init(String code) {
        System.out.println("创建节点对象");
        mutex = new InterProcessMutex(client, code);
    }

    /**
     * 加锁:互斥锁
     */
    public void lock(){
        try {
            mutex.acquire();
        } catch (Exception e) {
            e.printStackTrace();
            System.out.println("枷锁失败");
        }
    }

    /**
     * 释放锁
     */
    public void unLock() {
        try {
            mutex.release();
        } catch (Exception e) {
            e.printStackTrace();
            System.out.println("释放锁失败");
        }
    }
}

/**
 * @author LuGuoZheng
 * @date 2019/5/15 15:06
 * @Version 3.0
 * @description
 */
public class TestLock extends ZKLock {
    public TestLock(){
        super.init("/LGZ_TEST_CODE");
    }

    @Test
    public void test() throws InterruptedException {
        TestLock testLock = new TestLock();
        try {
            testLock.lock();
            System.out.println("");
            Thread.sleep(5000);
        } finally {
            testLock.unLock();
            System.out.println("释放成功");
        }
    }
}
