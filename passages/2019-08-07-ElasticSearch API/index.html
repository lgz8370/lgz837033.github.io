# 1. 创建索引
public class ElasticsearchRestClient {
    private static final Logger logger = LogManager.getLogger(ElasticsearchRestClient.class);
    @Before
    public void beforGet() {
        client = new RestHighLevelClient(
                RestClient.builder(new HttpHost("106.14.116.156", 9200, "http")));
        logger.info("ES客户端初始化");
    }
    
    @After
    public void afterGet() throws IOException {
        client.close();
        logger.info("关闭ES连接");
    }
    
    /**
     * 创建索引
     *
     * @throws IOException
     */
    @Test
    public void createIndex() throws IOException {
        GetIndexRequest getIndexRequest = new GetIndexRequest("item");
        // 索引是否存在
        boolean exists = client.indices().exists(getIndexRequest, RequestOptions.DEFAULT);
        if (!exists) {
            CreateIndexRequest request = new CreateIndexRequest("item");
            XContentBuilder builder = XContentFactory.jsonBuilder();
            builder.startObject();
            {
                // 固定格式
                builder.startObject("properties");
                {
                    // itemCode字段
                    builder.startObject("itemCode");
                    {
                        // 字段属性
                        builder.field("type", "keyword");
                    }
                    builder.endObject();

                    // itemDesc字段
                    builder.startObject("itemDesc");
                    {
                        // 字段属性
                        builder.field("type", "keyword");
                    }
                    builder.endObject();
                }
                builder.endObject();
            }
            builder.endObject();

            request.mapping("doc", builder);
            CreateIndexResponse createIndexResponse = client.indices().create(request, RequestOptions.DEFAULT);
            logger.info("创建索引结果:{}", createIndexResponse.isAcknowledged());
        } else {
            logger.info("[employee]索引已经存在");
        }
    }
}

# 2.新增数据
/**
     * 新增数据
     *
     * @param client
     * @param index   索引
     * @param id   文档id
     * @param jsonMap 需要建立索引的数据 key:字段,value:值
     */
    public static void index(RestHighLevelClient client, String index,String id, Map<String, Object> jsonMap) {
        IndexRequest indexRequest = new IndexRequest(index);
        // id为空则自动生成id
        if (StringUtil.isNotEmpty(id)) {
            indexRequest.id(id);
        }
        indexRequest.source(jsonMap);
        try {
            IndexResponse response = client.index(indexRequest, RequestOptions.DEFAULT);
            logger.info("索引结果:{}", response.status().getStatus());
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

# 3.删除数据
/**
     * 删除数据
     * 
     * @param client
     * @param index 索引名称
     * @param id 文档id
     */
    @Test
    public static void delete(RestHighLevelClient client, String index,String id) {
        DeleteRequest request = new DeleteRequest(index, id);
        DeleteResponse deleteResponse = null;
        try {
            deleteResponse = client.delete(request, RequestOptions.DEFAULT);
        } catch (Exception e) {
            e.printStackTrace();
            logger.warn("ES删除异常信息:{}",e.getMessage());
        }
        logger.warn("删除结果:{}",deleteResponse.status().getStatus());
    }

# 4.更新数据
/**
     *
     * @param client
     * @param index 索引名称
     * @param id 文档id
     */
    @Test
    public static void update(RestHighLevelClient client, String index,String id,Map<String,Object> map) {
        UpdateRequest request = new UpdateRequest(index,id);
        request.doc(map);
        UpdateResponse updateResponse = null;
        try {
            updateResponse = client.update(
                    request, RequestOptions.DEFAULT);
        } catch (IOException e) {
            e.printStackTrace();
        }
        logger.warn("ES更新结果:{}",updateResponse.getResult());
    }
